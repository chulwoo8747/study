function solution(numer1, denom1, numer2, denom2) {
    let mo = denom2 * denom1;
    let ja = numer1 * denom2 + numer2 * denom1;
    let a = mo;
    let b = ja;
    let gcd = 1;

    for (i = 2; i<=Math.min(mo,ja); i++) {
        if (mo %i === 0 && ja %i === 0){
            gcd = i;
        }
    }

    return [ja / gcd ,mo / gcd];
