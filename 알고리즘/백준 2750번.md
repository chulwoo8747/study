```c
#include<stdio.h>
int main() {
    int a, arr[1000] = { 0 };
    int tmp;
    scanf("%d", &a);
    for (int i = 0; i < a; i++) {
        scanf("%d", &arr[i]);
    }
    for (int i = 0; i < a; i++) {
        for (int j = 0; j < a - 1; j++) {
            if (arr[j + 1] < arr[j]) {
                tmp = arr[j + 1];
                arr[j + 1] = arr[j];
                arr[j] = tmp;
                }
        }
    }
    for (int i = 0; i < a; i++) {
        printf("%d\n", arr[i]);
    }
}
```
